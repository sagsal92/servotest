

#include <Servo.h> 
 
Servo myservo;  
int tpos = 10;    
 int pos=0;
void setup() 
{ 
  Serial.begin(9600);
  myservo.attach(4); 
   myservo.write(90);
} 
 
 
void loop() 
{ 
  if(Serial.available()>0)
  {
      tpos=Serial.parseInt();
      if(pos<tpos)//increesing degree
      {
        while(pos!=tpos)
        {
            myservo.write(pos);
            pos++;
            delay(15);
       }
    }
   else// decreesing degree
   {
      while(pos!=tpos)
      {
        myservo.write(pos);
        pos--;
        delay(15);
      }
    }
               // waits 15ms for the servo to reach the position 
  }                                  
}
